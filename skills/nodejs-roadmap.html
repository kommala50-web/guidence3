1. Introduction to Node.js

Description: Node.js is a JavaScript runtime built on Chrome's V8 engine for building server-side and network applications. It's event-driven and non-blocking.

// hello.js

6. Callbacks & Async Patterns

Description: Node initially used callbacks for async code ‚Äî understand callback style and callback hell and patterns to avoid it.

});

7. Promises & async/await

Description: Modern async patterns: Promises, chaining, and async/await for readable asynchronous code and error handling.

const fs = require('fs').promises;

async function read() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch(e) {
    console.error(e);
  }
}
read();

21. Authentication & Authorization

Description: Implement authentication with sessions (express-session) or stateless JWT. Protect routes via middleware.

const jwt = require('jsonwebtoken');
const token = jwt.sign({id:123}, 'secret', {expiresIn:'1h'});
// verify middleware
function auth(req,res,next){
  const token = req.headers.authorization?.split(' ')[1];
  if(!token) return res.sendStatus(401);
  jwt.verify(token,'secret',(err,user)=>{ if(err) return res.sendStatus(403); req.user=user; next(); });
}
